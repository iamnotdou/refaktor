## Wallet Hook (wagmi v2)

```typescript
// hooks/useFatura.ts
import { useWriteContract, useReadContract } from 'wagmi';
import { invoiceNFTAbi, contracts } from '@/lib/contracts';

export function useMintInvoice() {
  return useWriteContract();
}

export function useInvoice(tokenId: number) {
  return useReadContract({
    address: contracts.invoiceNFT,
    abi: invoiceNFTAbi,
    functionName: 'getInvoice',
    args: [BigInt(tokenId)],
  });
}
```

## Server State — TanStack Query

```typescript
// hooks/useMarketplace.ts
export function useMarketplace(filters: Filters) {
  return useQuery({
    queryKey: ['marketplace', filters],
    queryFn: () => api.get('/marketplace', { params: filters }),
    refetchInterval: 5000, // poll every 5s
  });
}
```

## Real-time Order Book

WebSocket ile order book updates:
```typescript
// hooks/useOrderBook.ts
export function useOrderBook(invoiceId: string) {
  const [book, setBook] = useState<OrderBook | null>(null);
  
  useEffect(() => {
    const socket = io(BACKEND_URL);
    socket.emit('subscribe:orderbook', invoiceId);
    socket.on('orderbook:update', setBook);
    
    return () => socket.disconnect();
  }, [invoiceId]);
  
  return book;
}
```

## State Management

| State | Yer | Neden |
|---|---|---|
| Wallet | wagmi (otomatik) | Built-in |
| Order book | TanStack Query + WebSocket | Server state, realtime |
| User session | Zustand | Persist localStorage |
| Form state | react-hook-form + zod | Validasyon |
| Modal/UI | local useState | Kısa ömürlü |

## Hata Senaryoları

| Senaryo | Frontend davranışı |
|---|---|
| Wallet bağlı değil | Disable buy/sell buttons, "Connect" CTA |
| Yanlış network | Modal: "Switch to Polygon Amoy" |
| Yetersiz USDC | Toast error + "Get test USDC" link |
| Tx reverted | Toast + sebep (decoded error) |
| Backend offline | Stale data badge + retry banner |
