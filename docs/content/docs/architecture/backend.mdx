# Backend Mimarisi

Backend'in 3 ana sorumluluğu var:
1. **REST API** — frontend'e veri sunmak
2. **Matching Engine** — bid/ask eşleştirip kontrata göndermek
3. **Event Listener** — kontrat eventlerini DB'ye yansıtmak

## Matching Engine

`backend/src/matching/engine.ts`:

```typescript
class MatchingEngine {
  private books = new Map<string, OrderBook>();
  
  async placeOrder(order: Order) {
    const book = this.getBook(order.invoiceId);
    book.add(order);
    
    while (this.canMatch(book)) {
      const trade = this.match(book);
      await this.settle(trade);
      this.broadcast(book);
    }
    
    await this.persist(book);
  }
  
  private canMatch(book: OrderBook) {
    const bestBid = book.bids[0];
    const bestAsk = book.asks[0];
    return bestBid && bestAsk && bestBid.price >= bestAsk.price;
  }
  
  private match(book: OrderBook) {
    const bid = book.bids[0];
    const ask = book.asks[0];
    const qty = Math.min(bid.remaining, ask.remaining);
    const price = ask.price; // maker price
    
    bid.remaining -= qty;
    ask.remaining -= qty;
    
    if (bid.remaining === 0) book.bids.shift();
    if (ask.remaining === 0) book.asks.shift();
    
    return { bid, ask, qty, price };
  }
  
  private async settle(trade: Trade) {
    // Submit to OrderBook.sol contract
    const tx = await this.contracts.orderBook.executeTrade(
      trade.invoiceId,
      trade.bid.maker,
      trade.ask.maker,
      trade.qty,
      parseUnits(trade.price.toString(), 6)
    );
    await tx.wait();
    
    await Trade.create({
      invoiceId: trade.invoiceId,
      buyer: trade.bid.maker,
      seller: trade.ask.maker,
      qty: trade.qty,
      price: trade.price,
      txHash: tx.hash,
      timestamp: new Date(),
    });
  }
}
```

export async function verifyInvoice(input: GibInput): Promise<GibResult> {
